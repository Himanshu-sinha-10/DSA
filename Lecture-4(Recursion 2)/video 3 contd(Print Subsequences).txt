#include <iostream>
using namespace std;

void print_subs(string input, string output) {
	if (input.length() == 0) {  // base case is when input becomes empty whatever output holds we will print that
		cout << output << endl;
		return;
	}
                                            // take decisions for recursive calls
	print_subs(input.substr(1), output); // lets not include the first character in the output string 
	print_subs(input.substr(1), output + input[0]); // lets include the first character in the output string
}

int main() {
	string input;
	cin >> input;
	string output = "";
	print_subs(input, output);
}
